 <script type="text/javascript">
   function initMap() {
     var latLng = new google.maps.LatLng(39.747805, -76.84716);
     var homeLatLng = new google.maps.LatLng(40.47805, -77.84716);

     var map = new google.maps.Map(document.getElementById('map'), {
       zoom: 7,
       center: latLng,
       mapTypeId: google.maps.MapTypeId.ROADMAP
     });

     var marker001 = new MarkerWithLabel({
       position: homeLatLng,
       draggable: true,
       raiseOnDrag: true,
       map: map,
       labelContent: "$425K",
       labelAnchor: new google.maps.Point(22, 0),
       labelClass: "labels", // the CSS class for the label
       labelStyle: {opacity: 0.75},
       icon: {}
     });

     var marker002 = new MarkerWithLabel({
       position: new google.maps.LatLng(40.475, -77.684),
       draggable: true,
       raiseOnDrag: true,
       map: map,
       labelContent: "$395K",
       labelAnchor: new google.maps.Point(22, 0),
       labelClass: "labels", // the CSS class for the label
       labelStyle: {opacity: 1.0}
     });



   }
 </script>
